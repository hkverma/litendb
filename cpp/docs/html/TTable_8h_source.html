269 TStatus TTable::CreateColumnLookUp(int64_t cnum,
270  std::shared_ptr<TColumn> col,
271  std::shared_ptr<TSchema> parentSchema,
272  std::shared_ptr<arrow::Field> parentField)
273 {
274 
275  if (nullptr != parentRowIdLookup_[cnum])
276  {
277  TLOG(INFO) << "Trying to rereate Join. child table=" << GetName() << " col=" << col->GetName();
278  return TStatus::OK();
279  }
280 
281  // Get parent column
282  auto ttable = parentSchema->GetTable();
283  if (!ttable)
284  {
285  return TStatus::Invalid("No table for parent schema.");
286  }
287 
288  auto parentColResult = ttable->GetColumn(parentField);
289  if (!parentColResult.ok())
290  return std::move(parentColResult.status());
291  auto parentCol = parentColResult.ValueOrDie();
292  if (parentCol->GetMap().ok())
293  {
294  parentCol->CreateReverseMap();
295  }
296 
297  // For each block, create a vector to look at the parent value
298  auto parentRowIdLookup =
299  std::make_shared<std::vector<std::vector<TRowId>>>(col->NumBlocks());
300 
301  for (auto blkNum=0; blkNum<col->NumBlocks(); blkNum++)
302  {
303  std::shared_ptr<arrow::NumericArray<ValueType>> arr =
304  std::dynamic_pointer_cast<arrow::NumericArray<ValueType>>(col->GetBlock(blkNum)->GetArray());
305  if (nullptr == arr)
306  {
307  return TStatus::Invalid("Non numeric array not supported in Column lookups");
308  }
309 
310  std::vector<TRowId> parentRowId(arr->length());
311 
312  for (auto rowNum=0; rowNum<arr->length(); rowNum++)
313  {
314  Type value = std::move(arr->Value(rowNum));
315  parentRowId[rowNum] = std::move(parentCol->GetRowId<Type,ArrayType>(value));
316  }
317  parentRowIdLookup->at(blkNum) = std::move(parentRowId);
318  }
319 
320  // Now set the parent Column
321  parentColumn_[cnum] = parentCol;
322  parentRowIdLookup_[cnum] = parentRowIdLookup;
323 
324  TLOG(INFO) << "Joined child table=" << GetName() << " col=" << col->GetName()
