10 // or it may indicate an error with an associated error message.

344 bool TStatus::Equals(const TStatus& s) const {
345  if (state_ == s.state_) {
346  return true;
347  }
348 
349  if (ok() || s.ok()) {
350  return false;
351  }
352 
353  if (detail() != s.detail()) {
354  if ((detail() && !s.detail()) || (!detail() && s.detail())) {
355  return false;
356  }
357  return *detail() == *s.detail();
358  }
359
